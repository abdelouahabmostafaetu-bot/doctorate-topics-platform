
// ===== Mathora — مساعد الشاشة الرئيسية: حد 50 رسالة / 4 ساعات =====
model AssistantUsage {
  id          String   @id @default(auto()) @map("_id") @db.ObjectId
  userId      String   @unique
  windowStart DateTime @default(now())
  count       Int      @default(0)

  @@map("assistant_usage")
}

// ===== نصائح الشريط العلوي — يتحكم فيها الأدمن من /admin/tips =====
model SiteTip {
  id        String   @id @default(auto()) @map("_id") @db.ObjectId
  text      String
  href      String?
  cta       String?
  active    Boolean  @default(true)
  order     Int      @default(0)
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  @@map("site_tips")
}
